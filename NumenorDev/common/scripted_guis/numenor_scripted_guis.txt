numenor_council_of_the_sceptre_hud = {
	is_valid = {
		OR = {
			has_title = title:e_numenor
			has_title = title:d_nindamos
			has_title = title:d_hyarnudore
			has_title = title:d_sorontil
			has_title = title:d_formenore
			has_title = title:d_andunie
			has_title = title:d_nisimaldar
			has_title = title:d_nunduine
			has_title = title:d_ciryanore
		}
		culture = culture:adunaic
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:disband_numenor_council
			}
		}
	}
}

numenor_council_member_check_opinion_of_numenor = {
	effect = {
		save_scope_as = numenor_ruler
		every_in_global_list = {
			variable = council_of_numenor
			if = {
				limit = { exists = title:e_numenor.holder }
				if = {
					limit = {
						opinion = {
							target = title:e_numenor.holder
							value > 75
						}
					}
					set_variable = {
						name = agrees_with_numenor
						value = 1
					}
				} else = {
					set_variable = {
						name = agrees_with_numenor
						value = 0
					}
				}
			}
		}
	}
}

numneor_variables_on_council_members = {
	every_in_global_list = {
		variable = council_of_numenor
		if = {
			limit = { has_variable = agrees_with_numenor }
			remove_variable = agrees_with_numenor
		}
	}
}
