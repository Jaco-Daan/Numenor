### Natural Disasters ###
numenor_spawn_earthquake_in_region_effect = {
	random_county_in_region = {
		region = $REGION$
		numenor_spawn_earthquake_in_county_effect = {
			COUNTY = this
			REGION = $REGION$
		}
	}
}

numenor_spawn_earthquake_in_county_effect = {
	$COUNTY$ = { save_scope_as = epicenter_county }

	# Track globally for exclusivity and distribution
	start_situation = {
		type = natural_disaster_earthquake
		save_scope_as = situation
	}

	scope:situation ?= {
		set_situation_center_province = scope:epicenter_county.title_province
		add_to_global_variable_list = {
			name = ongoing_earthquakes
			target = this
		}
		set_variable = {
			name = fatalities
			value = 0
		}

		set_variable = {
			name = severity
			value = 3
		}

		set_variable = {
			name = epicenter_county
			value = scope:epicenter_county
		}

		situation_sub_region:disaster_area = { add_dejure_title_to_sub_region = scope:epicenter_county }
		# Add neighbours to sub_region
		every_county_in_region = {
			region = $REGION$
			save_temporary_scope_as = neighbouring_county
			scope:situation.situation_sub_region:disaster_area = { add_dejure_title_to_sub_region = scope:neighbouring_county }
		}
	}
}

numenor_spawn_flood_in_region_effect = {
	$REGION$ = { save_scope_as = river_region }

	random_county_in_region = {
		region = $REGION$
		limit = { is_coastal_county = yes }
		spawn_flood_in_county_effect = {
			COUNTY = this
			REGION = $REGION$
		}
	}
}

numenor_spawn_flood_in_county_effect = {
	$COUNTY$ = { save_scope_as = flood_county }

	start_situation = {
		type = natural_disaster_flood
		save_scope_as = situation
	}

	# Track globally for exclusivity and distribution
	scope:situation ?= {
		set_situation_center_province = scope:flood_county.title_province
		add_to_global_variable_list = {
			name = ongoing_floods
			target = this
		}

		# Save region for exclusivity checks and Situation naming
		set_variable = {
			name = river_region
			value = scope:river_region
		}
		
		set_variable = {
			name = fatalities
			value = 0
		}

		set_variable = {
			name = severity
			value = 3
		}

		random_situation_sub_region = { add_geographical_region = scope:river_region }
		every_county_in_region = {
			region = $REGION$
			limit = { is_coastal_county = yes }
			save_temporary_scope_as = neighbouring_county
			scope:situation = { random_situation_sub_region = { add_dejure_title_to_sub_region = scope:neighbouring_county } }
		}
		log_natural_disaster = "started"
	}
}

numenor_define_natural_disaster_regions = {
	add_to_global_variable_list = { name = flood_region_list target = geographical_region:numenor_river_region }
}

# Counting evil rulers in a subregion 
check_evil_rulers_per_subregion_effect = {
    situation_sub_region:rivendell_lorien = {
        if = {
			limit = {
				any_situation_sub_region_county = {
					this.holder = { faith_is_evil = yes }
				}
			}
			set_variable = {
				name = evil_counties_per_subregion
				value = 0 
			}
			every_situation_sub_region_county = {
				limit = {
					holder = { faith_is_evil = yes }
				}
				add_to_list = evil_counties_in_lorien
			}
			every_in_list = {
				list = evil_counties_in_lorien
				PREV = {
					change_variable = {
						name = evil_counties_per_subregion
						add = 1
					}
				}
			}
			if = {
				limit = { var:evil_counties_per_subregion > 0 }
				if = {
					limit = { any_situation_sub_region_participant = { is_ai = no } }
					ordered_situation_sub_region_participant = {
						limit = { 
							is_ai = no 
							is_landed = yes
						}
						trigger_event = elven_hope.0030 # Display the amount of evil rulers and reduce hope accordingly 
						# trigger_event = { # Fire an event that resets these variables after 2 days 
							# id = elven_hope.0031 
							# days = 2
						# }
					}
				}
				else = {
					ordered_situation_sub_region_participant = {
						limit = {
							is_landed = yes 
						}
						trigger_event = elven_hope.0030 # Display the amount of evil rulers and reduce hope accordingly 
						# trigger_event = { # Fire an event that resets these variables after 2 days 
							# id = elven_hope.0031 
							# days = 2
						# }
					}
				}
			}
		}
	}
