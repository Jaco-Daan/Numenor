council_of_numenor_total_approval = {
	value = 0

	situation:numenor_situation ?= {
		every_participant_group = {
			limit = { participant_group_type = numenor_council_of_the_sceptre }
			every_situation_group_participant = {
				if = {
					limit = {
						OR = {
							opinion = {
								target = title:e_numenor.holder
								value > 75
							}
							has_variable = council_member_bribed
						}
						save_temporary_scope_value_as = {
							name = opinion_of_numenor_ruler
							value = 1
						}
					}
				}
				else_if = {
					limit = {
						save_temporary_scope_value_as = {
							name = opinion_of_numenor_ruler
							value = 0
						}
					}
				}
				add = { value = scope:opinion_of_numenor_ruler }
			}
		}
	}
}

council_of_numenor_total_approval_gui = {
	value = 0 

	if = {
		limit = { council_of_numenor_total_approval > 8 }
		add = 8
	}
	else_if = {
		limit = { council_of_numenor_total_approval < 0 }
		add = 0
	}
	else = {
		add = council_of_numenor_total_approval
	}
}

bribe_council_member_cost = {
	value = 0
	if = {
		limit = {
			exists = scope:recipient
			exists = scope:actor
		}
		if = {
			limit = {
				scope:recipient = {
					opinion = {
						target = scope:actor
						value < 1
					}
				}
			}
			add = 150
		}
		if = {
			limit = {
				scope:recipient = {
					opinion = {
						target = scope:actor
						value < -50
					}
				}
			}
			add = 150
		}
	}
}
