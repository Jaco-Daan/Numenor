numenor_on_title_gain = {
	effect = {
		if = {
			limit = {
				scope:title = title:d_nilulonde
				dynasty = dynasty:dynasty_tumakveh
				age > 15
			}
			add_opinion = {
				modifier = bellakar_hates_numenor
				target = title:e_numenor.holder
			}
		}

		if = {
			limit = {
				OR = {
					scope:title = title:d_andunie
					scope:title = title:d_nisimaldar
					scope:title = title:d_nunduine
					scope:title = title:d_ciryanore
					scope:title = title:d_nindamos
					scope:title = title:d_hyarnudore
					scope:title = title:d_sorontil
					scope:title = title:d_formenore
				}
				NOT = {
					any_in_global_list = {
						variable = council_of_numenor
						this = root
					}
				}
				NOT = { has_title = title:e_numenor }
			}
			if = {
				limit = { culture = culture:adunaic }
				add_to_global_variable_list = {
					name = council_of_numenor
					target = root
				}
			}
			remove_list_global_variable = {
				name = council_of_numenor
				target = scope:previous_holder
			}
		}

		if = {
			limit = {
				has_title = title:e_numenor
				NOT = {
					scope:previous_holder = {
						any_held_title = {
							tier = tier_duchy
							OR = {
								this = title:d_andunie
								this = title:d_nisimaldar
								this = title:d_nunduine
								this = title:d_ciryanore
								this = title:d_nindamos
								this = title:d_hyarnudore
								this = title:d_sorontil
								this = title:d_formenore
							}
						}
					}
				}
			}
			remove_list_global_variable = {
				name = council_of_numenor
				target = scope:previous_holder
			}
		}
	}
}
