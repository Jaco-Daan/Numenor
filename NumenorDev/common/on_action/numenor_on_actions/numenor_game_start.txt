numenor_on_game_start = {
    events = {
		numenor_historical_artifacts.0001
	}
	effect = {
		numenor_history_changes = yes
        every_living_character = {
			limit = {
				culture = culture:adunaic
				location ?= { geographical_region = middleearth_numenor }
			}
			# Removes all lower than 6 7
			remove_trait = blood_of_numenor_1
			remove_trait = blood_of_numenor_2
			remove_trait = blood_of_numenor_3
			remove_trait = blood_of_numenor_4
			remove_trait = blood_of_numenor_5

			add_trait = blood_of_numenor_7
		}
		set_up_council_of_sceptre = yes
		numenor_fire_intro_events = yes
		
		# Setup plague resistance
		every_county_in_region = {
			region = middleearth_numenor
			add_county_modifier = blessed_land_of_numenor_modifier
		}

		start_situation = {
			type = numenor_situation
			start_phase = numenor_situation_isolation
		}
		define_numenor_resources = yes
    }
}
