}

	NOT = {
		is_target_in_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:reclaim_numenorean_umbar_decision
		}
	}
	NOT = { title:e_umbar.hegemony = title:h_numenor }
}

numenor_in_isolation_phase = {
	trigger_if = {
		limit = { exists = situation:numenor_situation }
		situation:numenor_situation ?= { situation_current_phase = numenor_situation_isolation }
	}
}

numenor_can_declare_war = {
	trigger_if = {
		limit = {
			scope:actor = title:e_numenor.holder
			NOT = {
				is_target_in_global_variable_list = {
					name = unavailable_unique_decisions
					target = flag:disband_numenor_council
				}
			}
		}
		council_of_numenor_total_approval > total_council_members_on_numenor_council.75_percentage_mult
	}
}
