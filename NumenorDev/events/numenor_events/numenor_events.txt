numenor.3000 = {
	title = numenor.3000.t
	desc = numenor.3000.desc
	theme = stewardship_duty_focus
    content_source = realms_dlc
	override_background = {
		reference = docks_adunai
	}
	right_portrait = { 
		character = scope:ruler_of_numenor
		animation = personality_bold
	}
	left_portrait = { 
		character = root
		animation = personality_cynical
	}

	immediate = {
		if = {
			limit = {
				exists = title:k_bellazen.holder
				title:k_bellazen.holder.dynasty = dynasty:dynasty_isilitoreth
				title:k_bellazen.holder.primary_title = title:k_bellazen
			}
			title:k_bellazen.holder = { save_scope_as = korlea_vassal_decision }
		}
		
		else_if = { 
			limit = {
				exists = title:d_korlea.holder
				title:d_korlea.holder.dynasty = dynasty:dynasty_isilitoreth
				title:d_korlea.holder.primary_title = title:d_korlea
			}
			title:d_korlea.holder = { save_scope_as = korlea_vassal_decision } 
		}
	}

	option = { # Accept vassalage offer
		name = numenor.3000.a
		custom_tooltip = numenor.3000.a.tt

		create_title_and_vassal_change = {
            type = swear_fealty
            save_scope_as = change
        }
        change_liege = {
            liege = scope:ruler_of_numenor
            change = scope:change 
        }
		resolve_title_and_vassal_change = scope:change

		set_character_faith_with_conversion = faith:faith_adunaic
		
		title:e_bellakar.holder ?= {
			add_pressed_claim = scope:ruler_of_numenor.primary_title
			trigger_event = numenor.3001
		}

		# add_to_global_variable_list = { # Bellakar hats numenor
		# 	name = korlea_vassalization
		# 	target = flag:option_a
		# }
		add_to_global_variable_list = { # Decision dissapears
			name = unavailable_unique_decisions
			target = flag:decision_numenor_annex_korlea
		}

		ai_chance = {
			base = 60
		}
	}
 
	option = { # Reject Vassalage offer
		name = numenor.3000.b
		custom_tooltip = numenor.3000.b.tt

		ai_chance = {
			base = 0
		}
	}

	after = {
		title:e_bellakar.holder = {
			dynasty = {
				every_dynasty_member = {
					limit = { is_alive = yes }
					add_opinion = {
						modifier = bellakar_hates_numenor
						target = scope:ruler_of_numenor
					}
				}
			}
		}
	}
}
